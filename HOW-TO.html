Hızlı Başlangıç

Python API

from mockjutsu.core import jutsu

# Tek değer
jutsu.generate('tckn')
# → '45678901234'

# Locale ile
jutsu.generate('firstname', locale='DE')
# → 'Maximilian'

# Kart numarası
jutsu.generate('cardnum', network='amex')
# → '378282246310005'

# Balance aralık
jutsu.generate('balance', min=100, max=500)
# → 234.80

CLI

# Temel kullanım
mockjutsu generate tckn

# Locale ile
mockjutsu generate nationalid --locale US

# Kart ağı
mockjutsu generate cardnum --network mir

# Tüm parametreleri listele
mockjutsu list

# API sunucusunu başlat
mockjutsu serve --port 8000

Locale-Aware Örnekler

# taxid → locale'e göre değişir
jutsu.generate('taxid', locale='TR')  # VKN
jutsu.generate('taxid', locale='DE')  # USt-IdNr
jutsu.generate('taxid', locale='FR')  # SIREN

# employer_id
jutsu.generate('employer_id', locale='UK')  # CRN
jutsu.generate('employer_id', locale='FR')  # SIRET
jutsu.generate('employer_id', locale='RU')  # OGRN

bulk() — Binlerce Değer

# 1000 geçerli TCKN
ids = jutsu.bulk('tckn', count=1000)

# 500 Luhn-geçerli Visa kart
cards = jutsu.bulk('cardnum', count=500,
                   network='visa')

# 200 Almanya şirket adı
names = jutsu.bulk('company_name',
                   count=200, locale='DE')

template() — Yapısal Kayıtlar

records = jutsu.template({
  'id':        'uuid',
  'name':      'fullname',
  'card':      'cardnum',
  'amount':    'balance',
  'currency':  'currency',
  'timestamp': 'timestamp_iso',
}, count=100, locale='TR')
# → 100 kayıtlı liste, her biri 6 alanla

export() — Doğrudan DB Seed

schema = {'id':'uuid', 'name':'fullname',
           'card':'cardnum', 'balance':'balance'}

# SQL INSERT
sql = jutsu.export(schema, count=500,
  format='sql', table='users', locale='TR')

# CSV
csv = jutsu.export(schema, count=1000,
  format='csv')

# JSON
data = jutsu.export(schema, count=50,
  format='json')
